## 3. js/gui.js
const content = document.getElementById('content');

async function showProducts() {
    const res = await fetch('http://localhost:5000/api/products');
    const products = await res.json();
    let html = '<h2>Products</h2><table><tr><th>Name</th><th>Price</th><th>Stock</th></tr>';
    products.forEach(p => {
        html += `<tr><td>${p.name}</td><td>₹${p.price}</td><td>${p.stock}</td></tr>`;
    });
    html += '</table>';
    content.innerHTML = html;
}

async function showOrders() {
    const token = localStorage.getItem('token');
    if (!token) { alert('Please login as admin'); return; }
    const res = await fetch('http://localhost:5000/api/orders', {
        headers: { 'Authorization': `Bearer ${token}` }
    });
    const orders = await res.json();
    let html = '<h2>Orders</h2><table><tr><th>User</th><th>Products</th><th>Total</th><th>Status</th></tr>';
    orders.forEach(o => {
        const productList = o.products.map(p => `${p.product.name} (x${p.quantity})`).join(', ');
        html += `<tr><td>${o.user.name}</td><td>${productList}</td><td>₹${o.totalAmount}</td><td>${o.status}</td></tr>`;
    });
    html += '</table>';
    content.innerHTML = html;
}

async function showUsers() {
    const token = localStorage.getItem('token');
    if (!token) { alert('Please login as admin'); return; }
    const res = await fetch('http://localhost:5000/api/auth/users', {
        headers: { 'Authorization': `Bearer ${token}` }
    });
    const users = await res.json();
    let html = '<h2>Users</h2><table><tr><th>Name</th><th>Email</th></tr>';
    users.forEach(u => {
        html += `<tr><td>${u.name}</td><td>${u.email}</td></tr>`;
    });
    html += '</table>';
    content.innerHTML = html;
}
